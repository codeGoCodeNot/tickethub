// 1. Add this helper anywhere (e.g. top of account-dropdown.tsx)
const avatarGradient = (name: string) => {
  const colors = [
    "135deg,#f59e0b,#ef4444", "135deg,#06b6d4,#6366f1",
    "135deg,#10b981,#3b82f6", "135deg,#8b5cf6,#ec4899",
    "135deg,#f97316,#eab308", "135deg,#14b8a6,#6366f1",
  ];
  const i = [...name].reduce((acc, c) => acc + c.charCodeAt(0), 0) % colors.length;
  return `linear-gradient(${colors[i]})`;
};

// 2. Use it on AvatarFallback
<AvatarFallback style={{ background: avatarGradient(user.name ?? ""), color: "white", fontWeight: "700" }}>
  {user.name?.charAt(0).toUpperCase()}
</AvatarFallback>
